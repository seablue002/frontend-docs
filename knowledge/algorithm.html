                code 查看代码 
                var arr = [
    1,
    2,
    [
      11,
      22,
      33
    ],
    [
      [
        'a',
        'b'
      ]
    ]
  ]
  
  /**
   * arr 待处理数组
   * [deep] 要遍历多维数组的深度，默认值为1
   */
  function flat(arr, deep = 1) {
    if (deep <= 0) {
      return arr
    }
    var res = []
    for (let i = 0; i < arr.length; i++) {
      var item = arr[i]
      var type = Object.prototype.toString.call(item).slice(8, -1)
      if (type !== 'Array') {
        res.push(item)
      } else {
        res = res.concat(flat(item, deep - 1))
      }
    }

    return res
  }

  let res = flat(arr, 2)

  console.log(res) // [1, 2, 11, 22, 33, 'a', 'b']

function flat(arr, deep = 2) {
  return deep > 0 ?
    arr.reduce((prev, curr) => prev.concat(Array.isArray(curr) ? flat(curr, deep - 1): curr), [])
    : arr
}

let res = flat(arr, 2)

                # 数组排序

                数组排序的实现有多重途径，常用的算法有：冒泡排序、快速排序、选择排序、插入排序等

                • 冒泡排序 冒泡排序 (opens new window) 的英文名是 Bubble Sort ，它是一种比较简单直观的排序算法

                简单来说它会重复走访过要排序的数列，一次比较两个数，如果他们的顺序错误就会将他们交换过来，走访数列的工作是重复地进行直到没有再需要交换，也就是说该数列已经排序完成

                这个算法的名字由来是因为越小的元素会经由交换慢慢"浮"到数列的顶端 算法思路（以按递增顺序排列为例）：

                1、我们需要做一个内层循环来比较每对相邻元素的大小，如果前面大于后面，就让他们交换位置，我们要让小的数在前面，大的数在后面

                2、当内层循环结束时，在数组最后一位的元素，就一定是这个数组中最大的元素了，这时候除了最后一个元素不用再动以外（所以内层循环每循环一次就可以少循环一次）我们还要再来确定这个数组中第二大的元素，第三大的元素，以此类推，因此我们还需要一层外层循环。如果这个数组有 n 个元素我们就要确定 n - 1 个元素的位置，所以外层循环需要循环的次数就是 n - 1 次

                3、只需要内外两层循环嵌套，就可以把数组排序好

                参照2 (opens new window)、参照3 (opens new window)

                code 查看代码 
                var myArr = [89,34,76,15,98,25,67];  

function bubbleSort(arr) {
for (var i = 0; i < arr.length - 1; i++) {
    for (var j = 0; j < arr.length - i - 1; j++) {
        if(arr[j] > arr[j + 1]) {
            // 交换两个数的位置
            var temp = 0;
            temp = arr[j];
            arr[j] = arr[j + 1];
            arr[j + 1] = temp;
        }
    }
}    

    return  arr;      
}

console.log(bubbleSort(myArr));
                • 快速排序 排序算法 (opens new window)（Sorting algorithm）是计算机科学最古老、最基本的课题之一。要想成为合格的程序员，就必须理解和掌握各种排序算法。

                目前，最常见的排序算法大概有七八种，其中"快速排序"（Quicksort）使用得最广泛，速度也较快。它是图灵奖得主C. A. R. Hoare（1934--）于1960时提出来的。

                "快速排序"的思想很简单，整个排序过程只需要三步：

                1、在数据集中，选择一个元素作为基准

                2、所有小于"基准"的元素，都移到"基准"的左边；所有大于"基准"的元素，都移到"基准"的右边。

                3、对"基准"左边和右边的两个子集，不断重复第一步和第二步，直到所有子集只剩下一个元素为止。

                code 查看代码 
                var arr = [10, 1, 40, 55, 3, 90]

function quickSort (arr) {
  var len = arr.length
  if (len <= 1) {
    return arr
  }
  
  var pivotIndex = Math.floor(len / 2)
  var pivot = arr.splice(pivotIndex, 1)[0]

  var left = []
  var right = []
  for (let i = 0; i < arr.length; i++) {
    if (arr[i] <= pivot) {
      left.push(arr[i])
    } else if (arr[i] > pivot) {
      right.push(arr[i])
    }
  }

  return quickSort(left).concat([pivot], quickSort(right))
}

var sortedArr = quickSort(arr)
console.log(sortedArr) // [1, 3, 10, 40, 55, 90]

function fn (arr) {
  var len = arr.length
  
  while(len > 0) {
    var randomIdx = Math.floor(Math.random() * --len)
    // 交换i 与 randomIdx对应的数据
    var temp = arr[len]
    arr[len] = arr[randomIdx]
    arr[randomIdx] = temp
  }
}

                # 3. 找到第一个出现一次的字符？ 中级

                code 查看代码 
                var str = 'aaaceccbbbddjfffaaa'

function fn (str) {
  var idx = 0

  for (var i = 0; i < str.length; i++) {
    var count = 0
    for (var j = 0; j < str.length; j++) {
      if (str[i] === str[j]) {
        count++

        if (count >= 2) {
          break
        }
      }
    }

    if (count === 1) {
      idx = i
      break
    }
  }

  return idx
}

var index = fn(str)
console.log(index) // 4 'e'

                code 查看代码

function binarySearch (arr , target) {
  var start = 0,
  end = arr.length - 1
  while (end >= start) {
    middle = Math.floor((start + end) / 2)
    if (target === arr[middle]) {
      return middle
    } else if (target > arr[middle]) {
      start = middle + 1
    } else if (target < arr[middle]) {
      end = middle - 1
    }
  }
  return -1
}

var index = binarySearch(arr, 1)
console.log(index)

                # 5. 符号闭合匹配 中级

                code 查看代码 
                function fn(str) {
  var len = str.length
  var items = []
  if (len % 2 !== 0) {
    return false
  }

  for (let i = 0; i < len; i++) {
    var letter = items[items.length - 1]
    switch(str[i]) {
      case '(':
        items.push('(')
        break
      case '[':
        items.push('[')
        break
      case '{':
        items.push('{')
        break
      case ')':
        if (letter === '(') {
          items.pop()
        }
        break
      case ']':
        if (letter === '[') {
          items.pop()
        }
        break
      case '}':
        if (letter === '{') {
          items.pop()
        }
        break
    }
  }
  return items.length === 0
}

var str = '(){}[]'
console.log(fn(str))

                var compose = function(functions) {
  return function(x) {
    if (functions.length === 0) {

                # 7. 统计出现最多的字符次数 初级

                code 查看代码 
                var str = 'aaabcjkkkjjccc'

function fn(str) {
  var map = {}
  for (let i = 0; i < str.length; i++) {
    if (map[str[i]]) {
      map[str[i]] += 1
    } else {
      map[str[i]] = 1
    }
  }

  var res = Object.values(map).reduce((prev, cur, index) => {
    return prev[0] > cur ? prev : [cur, index]
  }, [0, -1])
  return [Object.keys(map)[res[1]], res[0]]
}


console.log(fn(str)) // ['c', 4]
